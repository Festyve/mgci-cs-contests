A — Bell Schedule Add

Time: O(1). Convert to minutes, add, mod 1440.

import sys

def main():
    t = sys.stdin.readline().strip()
    x = int(sys.stdin.readline())
    hh = int(t[:2]); mm = int(t[3:])
    total = (hh * 60 + mm + x) % 1440
    print(f"{total//60:02d}:{total%60:02d}")

if __name__ == "__main__":
    main()
B — Snowy Courtyard

Time: O(RC). Check each 2×2 block.

import sys

def main():
    R, C = map(int, sys.stdin.buffer.readline().split())
    grid = [sys.stdin.buffer.readline().strip().decode() for _ in range(R)]
    ans = 0
    for i in range(R - 1):
        for j in range(C - 1):
            if grid[i][j] == '.' and grid[i][j+1] == '.' and grid[i+1][j] == '.' and grid[i+1][j+1] == '.':
                ans += 1
    print(ans)

if __name__ == "__main__":
    main()
C — Library Next Taller Stack

Time: O(N). Monotonic decreasing stack from right to left.

import sys

def main():
    data = list(map(int, sys.stdin.buffer.read().split()))
    N = data[0]
    h = data[1:]
    ans = [0] * N
    st = []  # indices, heights strictly increasing (from top perspective)

    for i in range(N - 1, -1, -1):
        while st and h[st[-1]] <= h[i]:
            st.pop()
        ans[i] = (st[-1] + 1) if st else 0
        st.append(i)

    print(*ans)

if __name__ == "__main__":
    main()
D — Garneau Wi-Fi Coverage

Time: O(N log N). Greedy interval covering: always extend farthest among intervals starting before current.

import sys

def main():
    data = list(map(int, sys.stdin.buffer.read().split()))
    L, N = data[0], data[1]
    intervals = []
    idx = 2
    for _ in range(N):
        p = data[idx]; r = data[idx+1]; idx += 2
        l = max(1, p - r)
        rr = min(L, p + r)
        intervals.append((l, rr))

    intervals.sort()
    cur = 1
    i = 0
    best_r = 0
    used = 0

    while cur <= L:
        while i < N and intervals[i][0] <= cur:
            if intervals[i][1] > best_r:
                best_r = intervals[i][1]
            i += 1

        if best_r < cur:
            print(-1)
            return

        used += 1
        cur = best_r + 1

    print(used)

if __name__ == "__main__":
    main()
E — Locker Light Switch

Time: O(N). Greedy left-to-right + difference array for flip parity.

import sys

def main():
    data = sys.stdin.buffer.read().split()
    N = int(data[0]); K = int(data[1])
    S = data[2].decode()

    diff = [0] * (N + 1)
    flip = 0
    ops = 0

    for i in range(N):
        flip ^= diff[i]
        bit = (ord(S[i]) - 48) ^ flip
        if bit == 1:
            if i + K > N:
                print(-1)
                return
            ops += 1
            flip ^= 1
            diff[i + K] ^= 1

    print(ops)

if __name__ == "__main__":
    main()
